CREATE PROCEDURE sp_UpdatePatientPassword
    @UserId INT,
    @NewPassword NVARCHAR(MAX)
AS
BEGIN
    SET NOCOUNT ON;

    -- тглеп дсйсод (д-Hash) тбеш доиефм
    UPDATE Patients
    SET PasswordHash = @NewPassword
    WHERE Id = @UserId;

    -- озжйш 1 ан теглп, 0 ан ма роца ощъощ
    SELECT @@ROWCOUNT; 
END
GO
